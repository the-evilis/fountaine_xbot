# Функции валидации
def validate_name(text):
    cleaned = text.strip()
    pattern = r"^[а-яА-ЯёЁa-zA-Z\s]+$"
    return bool(re.match(pattern, cleaned)) and len(cleaned) > 0

def validate_phone(text):
    cleaned = re.sub(r'\D', '', text)
    return len(cleaned) >= 9

def validate_time(text):
    cleaned = text.strip()
    return len(cleaned) > 0

def validate_goal(text):
    cleaned = text.strip()
    return len(cleaned) > 0

async def confirm_registration(message: types.Message, state: FSMContext):
    data = await state.get_data()
    lang = data.get("language", "ru")
    
    saved_to_sheets = False
    
    # Сохраняем в Google Sheets
    if sheet:
        try:
            sheet.append_row([
                datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
                data.get('name', ''),
                data.get('phone', ''),
                data.get('level', ''),
                data.get('time', ''),
                'Telegram',
                'Новая',
                data.get('goal', '')
            ])
            logger.info(f"Registration saved for {data.get('name')}")
            saved_to_sheets = True
        except Exception as e:
            logger.error(f"Google Sheets error: {e}")
            await message.answer(f"{TEXTS[lang]['sheets_error']}: {str(e)}")
    else:
        logger.warning("Google Sheets not configured")
        await message.answer(TEXTS[lang]["sheets_not_configured"])
    
    # Уведомление админу
    if ADMIN_CHAT_ID:
        admin_text = f"""
{TEXTS[lang]["admin_notification"]}

{TEXTS[lang]["name"]}: {data.get('name')}
{TEXTS[lang]["phone"]}: {data.get('phone')}
{TEXTS[lang]["level"]}: {data.get('level')}
{TEXTS[lang]["time"]}: {data.get('time')}
{TEXTS[lang]["goal"]}: {data.get('goal')}
"""
        try:
            await message.bot.send_message(
                ADMIN_CHAT_ID,
                admin_text,
                reply_markup=InlineKeyboardMarkup(
                    inline_keyboard=[[InlineKeyboardButton(text=TEXTS[lang]["call_admin"], callback_data="call_admin")]]
                )
            )
            logger.info(f"Admin notified about registration from {message.from_user.id}")
        except Exception as e:
            logger.error(f"Failed to notify admin: {e}")
    
    await message.answer(TEXTS[lang]["success"])
    await message.answer(
        TEXTS[lang]["main_menu"],
        reply_markup=get_main_menu_keyboard(lang)
    )
    await state.set_state(RegistrationStates.main_menu)

async def handle_ask_question(message: types.Message, state: FSMContext):
    data = await state.get_data()
    lang = data.get("language", "ru")
    
    question_text = message.text.strip()
    
    # Валидация
    if not question_text or len(question_text) < 3:
        await message.answer(TEXTS[lang]["invalid_question"])
        return
    
    # Сохраняем вопрос в логи
    logger.info(f"Question from user {message.from_user.id}: {question_text}")
    
    # Уведомление админу
    if ADMIN_CHAT_ID:
        admin_text = f"""
{TEXTS[lang]["admin_question"]}

Пользователь: {message.from_user.first_name}
Вопрос: {question_text}
"""
        try:
            await message.bot.send_message(ADMIN_CHAT_ID, admin_text)
            logger.info(f"Admin notified about question from {message.from_user.id}")
        except Exception as e:
            logger.error(f"Failed to notify admin: {e}")
    
    await message.answer(TEXTS[lang]["question_received"])
    await message.answer(
        TEXTS[lang]["main_menu"],
        reply_markup=get_main_menu_keyboard(lang)
    )
    await state.set_state(RegistrationStates.main_menu)

async def handle_faq_callback(callback: types.CallbackQuery, state: FSMContext):
    data = await state.get_data()
    lang = data.get("language", "ru")
    
    faq_map = {
        "faq_1": ("faq_q1", "faq_a1"),
        "faq_2": ("faq_q2", "faq_a2"),
        "faq_3": ("faq_q3", "faq_a3"),
        "faq_4": ("faq_q4", "faq_a4"),
    }
    
    if callback.data in faq_map:
        q_key, a_key = faq_map[callback.data]
        answer = f"{TEXTS[lang][q_key]}\n\n{TEXTS[lang][a_key]}"
        await callback.message.answer(answer, reply_markup=get_faq_keyboard(lang))
    elif callback.data == "back_to_menu":
        await callback.message.answer(
            TEXTS[lang]["main_menu"],
            reply_markup=get_main_menu_keyboard(lang)
        )
        await state.set_state(RegistrationStates.main_menu)
    
    await callback.answer()
